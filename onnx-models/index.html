Quick Start: YOLOv8 Object Detection

This is the most common DNN use case. Full working script — copy, paste, run.

import cv2
import numpy as np

# Load the ONNX model
MODEL = "yolov8n.onnx"
net = cv2.dnn.readNet(MODEL)  # ENGINE_AUTO in v5
INPUT_SIZE = (640, 640)
CONF = 0.5
NMS = 0.45

# COCO class names (YOLOv8)
CLASSES = ["person","bicycle","car","motorcycle","airplane","bus","train","truck","boat",
           "traffic light","fire hydrant","stop sign","parking meter","bench","bird","cat",
           "dog","horse","sheep","cow","elephant","bear","zebra","giraffe","backpack",
           "umbrella","handbag","tie","suitcase","frisbee","skis","snowboard","sports ball",
           "kite","baseball bat","baseball glove","skateboard","surfboard","tennis racket",
           "bottle","wine glass","cup","fork","knife","spoon","bowl","banana","apple",
           "sandwich","orange","broccoli","carrot","hot dog","pizza","donut","cake","chair",
           "couch","potted plant","bed","dining table","toilet","tv","laptop","mouse",
           "remote","keyboard","cell phone","microwave","oven","toaster","sink",
           "refrigerator","book","clock","vase","scissors","teddy bear","hair drier","toothbrush"]

def detect(img_path, output_path="output.jpg"):
    img = cv2.imread(img_path)
    h, w = img.shape[:2]

    # Preprocess
    blob = cv2.dnn.blobFromImage(img, 1/255.0, INPUT_SIZE, (0,0,0), swapRB=True, crop=False)
    net.setInput(blob)

    # Inference
    outputs = net.forward()

    # Parse YOLOv8 output: [1, 84, 8400]
    # 84 = 4 (bbox) + 80 (COCO scores)
    output = outputs[0].transpose(1, 0)  # [8400, 84]

    boxes, scores, class_ids = [], [], []
    for row in output:
        cls_scores = row[4:]
        cls_id = np.argmax(cls_scores)
        max_score = cls_scores[cls_id]
        if max_score < CONF:
            continue
        cx, cy, bw, bh = row[:4]
        x1 = int((cx - bw / 2) * w / INPUT_SIZE[0])
        y1 = int((cy - bh / 2) * h / INPUT_SIZE[1])
        x2 = int((cx + bw / 2) * w / INPUT_SIZE[0])
        y2 = int((cy + bh / 2) * h / INPUT_SIZE[1])
        boxes.append([x1, y1, x2 - x1, y2 - y1])
        scores.append(float(max_score))
        class_ids.append(cls_id)

    # NMS
    indices = cv2.dnn.NMSBoxes(boxes, scores, CONF, NMS)

    # Draw
    for i in indices:
        x, y, bw, bh = boxes[i]
        label = f"{CLASSES[class_ids[i]]} {scores[i]:.2f}"
        cv2.rectangle(img, (x, y), (x + bw, y + bh), (0, 255, 0), 2)
        cv2.putText(img, label, (x, y - 5), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 1)

    cv2.imwrite(output_path, img)
    print(f"Saved to {output_path}")

detect("input.jpg")

Convert PyTorch → ONNX

Before OpenCV 5 can run your model, export it to ONNX:

# YOLOv8/YOLO11 (ultralytics)
from ultralytics import YOLO
model = YOLO("yolov8n.pt")
model.export(format="onnx", opset=12, simplify=True)

# Generic PyTorch model
import torch
model = YourModel()
model.load_state_dict(torch.load("weights.pt"))
model.eval()
dummy = torch.randn(1, 3, 640, 640)
torch.onnx.export(model, dummy, "model.onnx",
    input_names=["images"],
    output_names=["output"],
    opset_version=12,
    dynamic_axes={"images": {0: "batch"}})

Engine Selection — When to Use What

EngineBest For
ENGINE_AUTO (default)95% of cases. Tries new engine, falls back to classic.
ENGINE_CLASSICWhen you need CUDA/OpenCL GPU, or a model that fails on the new engine.
ENGINE_ORTONNX Runtime backend — GPU via CUDA/TensorRT execution providers.
ENGINE_NEWForce new engine. Fails hard if the model uses unsupported ops.

Real-Time Webcam Inference

def webcam_inference(model_path="yolov8n.onnx"):
    net = cv2.dnn.readNet(model_path)
    cap = cv2.VideoCapture(0)  # or video file path

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        blob = cv2.dnn.blobFromImage(frame, 1/255.0, (640, 640), (0,0,0), swapRB=True, crop=False)
        net.setInput(blob)
        outputs = net.forward()

        # ... parse outputs, draw boxes (same as above) ...

        cv2.imshow("OpenCV 5 DNN — YOLOv8", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

MobileNet SSD — Lightweight Detection

For edge devices and Raspberry Pi, MobileNet SSD is the go-to:

net = cv2.dnn.readNet("ssd_mobilenet_v1.onnx")

# COCO labels for MobileNet SSD
LABELS = ["background","aeroplane","bicycle","bird","boat","bottle","bus","car",
          "cat","chair","cow","diningtable","dog","horse","motorbike","person",
          "pottedplant","sheep","sofa","train","tvmonitor"]

# Preprocess — MobileNet SSD uses 300x300
blob = cv2.dnn.blobFromImage(img, 0.007843, (300, 300), (127.5, 127.5, 127.5), swapRB=False)
net.setInput(blob)
detections = net.forward()  # shape: [1, 1, N, 7]

for i in range(detections.shape[2]):
    confidence = detections[0, 0, i, 2]
    if confidence > 0.5:
        cls_id = int(detections[0, 0, i, 1])
        x1 = int(detections[0, 0, i, 3] * w)
        y1 = int(detections[0, 0, i, 4] * h)
        x2 = int(detections[0, 0, i, 5] * w)
        y2 = int(detections[0, 0, i, 6] * h)
        cv2.rectangle(img, (x1, y1), (x2, y2), (0, 255, 0), 2)
Heads-up: MobileNet SSD preprocessing uses different mean and swapRB=False. Don't copy-paste YOLO's blob settings — the preprocessing must match how the model was trained.

Async Inference — OpenCV 5 Only

The new engine supports non-blocking inference. Useful when you have other work between frames.

net.setInput(blob)
future = net.forwardAsync()       # returns immediately
# ... do other work while model runs ...
result = future.get()             # blocks until done

Debugging Inference Failures

SymptomFix
Model fails to loadTry ENGINE_CLASSIC — the model may use ops the new engine doesn't support yet
All detections are wrongCheck swapRB, mean, and scalefactor match training
Output shape doesn't match expectedPrint output.shape — YOLO exports vary between ultralytics versions
Memory keeps growingDon't re-create the Net object per frame — create once, reuse
GPU not being usedNew engine is CPU-only. Switch to ENGINE_CLASSIC + CUDA backend
